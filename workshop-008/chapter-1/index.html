Python / asyncio
import asyncio, aiohttp

async def fetch_student(session, sid):
    async with session.get(f'/api/students/{sid}') as r:
        return await r.json()

async def fetch_all(ids):
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_student(session, i) for i in ids]
        results = await asyncio.gather(*tasks)
    for s in results:
        print(s['name'])

asyncio.run(fetch_all([1, 2, 3]))

FastAPI 서버

Python / FastAPI
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import pandas as pd

app = FastAPI()

class AnalysisRequest(BaseModel):
    zip_code: str
    bedrooms: int = 3

@app.post('/analyze')
async def analyze(req: AnalysisRequest):
    df = pd.read_csv('properties.csv')
    subset = df[
        (df['zip'] == req.zip_code) &
        (df['beds'] == req.bedrooms)
    ]
    if subset.empty:
        raise HTTPException(404, "No data")
    return {
        'zip': req.zip_code,
        'count': len(subset),
        'median': int(subset['price'].median()),
        'avg':    int(subset['price'].mean())
    }

비동기 크롤러

Python / asyncio + aiohttp
import asyncio
import aiohttp
from bs4 import BeautifulSoup

async def scrape_listing(session, url):
    async with session.get(url) as r:
        html = await r.text()
    soup = BeautifulSoup(html, 'html.parser')
    price = soup.select_one('.price').text.strip()
    return {'url': url, 'price': price}

async def scrape_all(urls):
    async with aiohttp.ClientSession() as session:
        tasks = [scrape_listing(session, u) for u in urls]
        return await asyncio.gather(*tasks)

urls = [
    'https://zillow.com/homedetails/1/',
    'https://zillow.com/homedetails/2/',
]
results = asyncio.run(scrape_all(urls))
